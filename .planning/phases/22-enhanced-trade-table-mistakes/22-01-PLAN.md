3. Create a `SortableHeader` component (inside TradeTable.tsx, not exported):
   ```typescript
   function SortableHeader({ colKey, label, sortKey, sortDir, onSort, onReorder }: {
     colKey: ColumnKey; label: string;
     sortKey: ColumnKey | null; sortDir: "asc" | "desc";
     onSort: (k: ColumnKey) => void;
     onReorder?: boolean; // show grip handle when true
   }) {
     const { attributes, listeners, setNodeRef, transform, transition, isDragging } = useSortable({ id: colKey });
     const style = { transform: CSS.Transform.toString(transform), transition, opacity: isDragging ? 0.5 : 1 };
     return (
       <th ref={setNodeRef} style={style} className="...existing th classes...">
         <div className="flex items-center gap-1">
           {onReorder && (
             <span {...attributes} {...listeners} className="p-0.5 rounded cursor-grab active:cursor-grabbing hover:dark:bg-slate-700 hover:bg-slate-200">
               <GripVertical className="w-3 h-3 dark:text-slate-500 text-slate-400" />
             </span>
           )}
           <button onClick={() => onSort(colKey)} className="...existing sort button classes...">
             {label} {sortKey === colKey ? (sortDir === "asc" ? "▲" : "▼") : ""}
           </button>
         </div>
       </th>
     );
   }
   ```
   IMPORTANT: The GripVertical icon gets `{...attributes} {...listeners}` — NOT the entire th or the sort button. This prevents drag from intercepting sort clicks (see Research pitfall 2).

4. In the main TradeTable component:
   - Set up sensors: `const sensors = useSensors(useSensor(PointerSensor, { activationConstraint: { distance: 5 } }), useSensor(KeyboardSensor));`

     ```typescript
     const handleColumnDragEnd = (event: DragEndEvent) => {
       const { active, over } = event;
       if (over && active.id !== over.id && visibleColumns) {
         const cols = visibleColumns ?? DEFAULT_COLUMNS;
         const oldIdx = cols.indexOf(active.id as ColumnKey);
         const newIdx = cols.indexOf(over.id as ColumnKey);
         onReorderColumns?.(arrayMove(cols, oldIdx, newIdx));
       }
     };
     ```
   - Wrap the `<thead>` content (the header `<tr>`) inside `<DndContext sensors={sensors} collisionDetection={closestCenter} onDragEnd={handleColumnDragEnd}>` and `<SortableContext items={headers.map(h => h.key)} strategy={horizontalListSortingStrategy}>`
