```typescript
function addSymbol(sym: string) {
  const upper = sym.toUpperCase().trim();
  if (!upper) return;
  if (symbols.includes(upper)) { setSearchValue(""); return; }
  if (symbols.length >= MAX_SYMBOLS) return;
  setSymbols((prev) => [...prev, upper]);
  setSearchValue("");
  setMatrix(null);
}
```
